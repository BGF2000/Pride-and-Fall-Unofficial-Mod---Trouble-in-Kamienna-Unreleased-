BGF_better_prepare = {
    if = {
		limit = { has_idea = BGF_a_long_road_ahead }
		remove_ideas = BGF_a_long_road_ahead
	}
	else_if = {
		limit = { has_idea = BGF_a_long_road_ahead_2 }
		swap_ideas = {
			remove_idea = BGF_a_long_road_ahead_2
			add_idea = BGF_a_long_road_ahead
		}
	}
	else_if = {
		limit = { has_idea = BGF_a_long_road_ahead_3 }
		swap_ideas = {
			remove_idea = BGF_a_long_road_ahead_3
			add_idea = BGF_a_long_road_ahead_2
		}
	}
	
}
